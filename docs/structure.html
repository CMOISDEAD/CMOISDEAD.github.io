package co.uniquindio.programacion.Perro;

import javax.swing.JOptionPane;

/**
 *
 * @author camilo
 */
public class Perro {

    public static void main(String[] args) {
        double areaCuadradoPrincipal = 100, areaTotal, areaRestante;
        double base, altura;
        double largo, ancho;
        double lado;

        double areaTriangulo, areaRectangulo, areaCuadrado;
        base = Double.parseDouble(JOptionPane.showInputDialog("Ingresa la base del triangulo: "));
        altura = Double.parseDouble(JOptionPane.showInputDialog("Ingresa la altura del triangulo: "));

        largo = Double.parseDouble(JOptionPane.showInputDialog("Ingresa la base del rectangulo: "));
        ancho = Double.parseDouble(JOptionPane.showInputDialog("Ingresa la altura del rectangulo: "));

        lado = Double.parseDouble(JOptionPane.showInputDialog("Ingresa el lado del cuadrado: "));

        areaTriangulo = CalcularAreaTriangulo(base, altura);
        areaRectangulo = CalcularAreaRectangulo(largo, ancho);
        areaCuadrado = CalcularAreaCuadrado(lado);
        areaTotal = CalcularAreaTotal(areaTriangulo, areaRectangulo, areaCuadrado);

        areaRestante = calcularAreaRestante(areaTotal, areaCuadradoPrincipal);

        Imprimir("El area restante es:"+areaRestante);
    }

    public static double CalcularAreaTriangulo(double base, double altura) {
        double area = (base * altura) / 2;
        return area;
    }

    public static double CalcularAreaRectangulo(double largo, double ancho) {
        double area = (largo * ancho);
        return area;
    }

    public static double CalcularAreaCuadrado(double lado) {
        double area = (lado * lado);
        return area;
    }

    public static double CalcularAreaTotal(double area1, double area2, double area3) {
        double areaTotal = area1 + area2 + area3;

        return areaTotal;
    }

    public static double calcularAreaRestante(double areaPerro, double areaCuadradoPrincipal) {
        double area = (areaPerro - areaCuadradoPrincipal);

        return area;
    }

    public static void Imprimir(String texto) {
        JOptionPane.showMessageDialog(null, texto);
    }
}
